# Pronoms et adjectifs indefinis

autre = ya nkaka  
tout = nionso  
quelqu'un = muntu mosi  
seul = mosi  
beaucoup = mingi  
chaque (personne) = muntu na muntu  
personne = muntu mosi ve  
moi-même = munu mosi  
aucun personne = muntu mosi ve  
aucun chose = kima mosi ve
